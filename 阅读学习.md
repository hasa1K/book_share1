1. for循环下使用switch和break，break只会跳出switch，不会跳出for循环，需要跟标签搭配使用，break标签，才能跳出for循环
   

```go

package main

import (
	"context"
	"fmt"
)

func listing1() {
	for i := 0; i < 5; i++ {
		fmt.Printf("%d ", i)

		switch i {
		default:
		case 2:
			break
		}
	}
}

func listing2() {
loop:
	for i := 0; i < 5; i++ {
		fmt.Printf("%d ", i)

		switch i {
		default:
		case 2:
			break loop
		}
	}
}

func listing3(ctx context.Context, ch <-chan int) {
	for {
		select {
		case <-ch:
			// Do something
		case <-ctx.Done():
			break
		}
	}
}

func listing4(ctx context.Context, ch <-chan int) {
loop:
	for {
		select {
		case <-ch:
			// Do something
		case <-ctx.Done():
			break loop
		}
	}
}```

2. for range中使用指针



```go
package main

import "fmt"

type Customer struct {
   ID      string
   Balance float64
}

type Store struct {
   m map[string]*Customer
}

func main() {
   s := Store{
      m: make(map[string]*Customer),
   }
   s.storeCustomers3([]Customer{
      {ID: "1", Balance: 10},
      {ID: "2", Balance: -10},
      {ID: "3", Balance: 0},
   })
   print(s.m)
}

func (s *Store) storeCustomers(customers []Customer) {
    // for range时会创造一个临时变量，customer的地址不是customers中变量的地址，而是临时变量的地址
   for _, customer := range customers {
      fmt.Printf("%p\n", &customer)
      s.m[customer.ID] = &customer
   }
}

func (s *Store) storeCustomers2(customers []Customer) {
   for _, customer := range customers {
       // 可以采用赋值给一个变量解决这个问题
      current := customer
      s.m[current.ID] = &current
   }
}


func print(m map[string]*Customer) {
   for k, v := range m {
      fmt.Printf("key=%s, value=%#v\n", k, v)
   }
}
